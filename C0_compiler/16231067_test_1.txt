int factorial(int n){
	int result;
	if(n == 1){
		result = 1;
	}
	else{
		result = n * factorial(n-1);
	}
	return(result);
	printf("Run success!");
}

	int j;
	int div,mod;
	j=2;div=0;mod=0;
	do{
		div=n/j;
		mod=n-div*j;
		j=j+1;	
	}while(mod!=0)
	if(j==n){
		return(true);
	}	
	else{
		return(false);
	}
}
