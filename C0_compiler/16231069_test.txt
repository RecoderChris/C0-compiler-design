const int max=100;
const char line='_';
int num_1,num_2,fibo[100];
void swap				
{
	int temp;
	temp=num_1;
	num_1=+num_2;
	num_2=temp;
	return;
	}
void fibonacci(int n)	
{
	fibo[n]=+fibo[+n-2]+fibo[n-1];
	return;
	}
int minus				
{
	int temp;
	temp=num_1-num_2;
	return(temp);
	}
int factorial(int n)	
{
	if(n==0)
		return(1);
	else
		return(n*factorial(n-1));
}

						c=2;
